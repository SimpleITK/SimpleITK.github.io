20""" A SimpleITK example demonstrating various image grid manipulation
21 techniques. """
22
23import sys
24import SimpleITK as sitk
25
26if len(sys.argv) < 3:
27 print("Usage: " + sys.argv[0] + " <input-1> <input-2>")
28 sys.exit(1)
29
30# Two vector images of same pixel type and dimension expected
31image_1 = sitk.ReadImage(sys.argv[1])
32image_2 = sitk.ReadImage(sys.argv[2])
33
34# Join two N-D Vector images to form an (N+1)-D image
35join = sitk.JoinSeriesImageFilter()
36joined_image = join.Execute(image_1, image_2)
37
38# Extract first three channels of joined image (assuming RGB)
39select = sitk.VectorIndexSelectionCastImageFilter()
40channel1_image = select.Execute(joined_image, 0, sitk.sitkUInt8)
41channel2_image = select.Execute(joined_image, 1, sitk.sitkUInt8)
42channel3_image = select.Execute(joined_image, 2, sitk.sitkUInt8)
43
44# Recompose image (should be same as joined_image)
45compose = sitk.ComposeImageFilter()
46composed_image = compose.Execute(channel1_image, channel2_image, channel3_image)
47
48# Select same subregion using image slicing operator
49sliced_image = composed_image[100:400, 100:400, 0]
50
51# Select same subregion using ExtractImageFilter
52extract = sitk.ExtractImageFilter()
53extract.SetSize([300, 300, 0])
54extract.SetIndex([100, 100, 0])
55extracted_image = extract.Execute(composed_image)
56
57# Select same sub-region using CropImageFilter (NOTE: CropImageFilter cannot
58# reduce dimensions unlike ExtractImageFilter, so cropped_image is a three
59# dimensional image with depth of 1)
60crop = sitk.CropImageFilter()
61crop.SetLowerBoundaryCropSize([100, 100, 0])
62crop.SetUpperBoundaryCropSize(
63 [composed_image.GetWidth() - 400, composed_image.GetHeight() - 400, 1]
64)
65cropped_image = crop.Execute(composed_image)
