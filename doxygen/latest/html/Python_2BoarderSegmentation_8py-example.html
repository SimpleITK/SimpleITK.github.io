20""" An example script demonstrating boarder segmentation in SimpleITK. """
21
22import sys
23import os
24import SimpleITK as sitk
25
26# verify that we have the correct number of arguments
27if len(sys.argv) != 5:
28 sys.stderr.write("Usage: {sys.argv[0]} inputFile outputFile replaceValue upperThreshold\n")
29 sys.exit(1)
30
31# copy the arguments in to variables
32inputFileName = sys.argv[1]
33outputFileName = sys.argv[2]
34replaceValue = int(sys.argv[3])
35upperThreshold = float(sys.argv[4])
36
37# Read the file into an sitkImage
38image = sitk.ReadImage(inputFileName)
39
40# Threshold the value [0,2), results in values inside the range 1, 0 otherwise
41boundary = sitk.BinaryThreshold(image, 0, upperThreshold, 1, 0)
42
43boundary = sitk.BinaryMorphologicalClosing(boundary, [1] * image.GetDimension())
44
45# Remove any label pixel not connected to the boarder
46boundary = sitk.BinaryGrindPeak(boundary)
47
48boundary = sitk.Cast(boundary, image.GetPixelID())
49
50# Multiply, the input image by not the boarder.
51# This will multiply the image by 0 or 1, where 0 is the
52# boarder. Making the board 0
53image = image * ~boundary
54
55# add the replace value to the pixel on the board
56image = image + (boundary * replaceValue)
57
58if "SITK_NOSHOW" not in os.environ:
59 sitk.Show(image, "Boarder Segmentation")
