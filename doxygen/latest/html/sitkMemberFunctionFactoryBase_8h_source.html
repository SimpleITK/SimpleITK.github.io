41 template <typename T> struct hash : public std::hash<T>{};

45 template <>
46 struct hash< std::pair<int, int> >
47  : public std::unary_function<std::pair<int,int>, std::size_t> {
48  std::size_t operator()( const std::pair<int, int > &p ) const
49  { return std::hash<size_t>()( size_t(p.first) * prime + p.second ); }
50 private:
51  static const std::size_t prime = 16777619u;
52 };
